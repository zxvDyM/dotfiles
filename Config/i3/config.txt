# ~/.dotfiles/Config/i3/config.txt

# Usar Nerd Font tamaño 15 para la fuente
font pango: Iosevka Nerd Font 15

# Modificador principal (Windows key)
set $mod Mod4

# Lanzar terminal
bindsym $mod+Return exec kitty

# Lanzar dmenu (launcher de aplicaciones)
bindsym $mod+d exec dmenu_run

# Cerrar ventana enfocada
bindsym $mod+Shift+q kill

# Reload configuración (útil para aplicar cambios sin reiniciar sesión)
bindsym $mod+Shift+c reload

# Restart i3 inplace
bindsym $mod+Shift+r restart

# Mover foco entre ventanas con Mod + h,j,k,l (vim style)
bindsym $mod+h focus left
bindsym $mod+j focus down
bindsym $mod+k focus up
bindsym $mod+l focus right

# Mover ventana enfocada con Mod + Shift + h,j,k,l
bindsym $mod+Shift+h move left
bindsym $mod+Shift+j move down
bindsym $mod+Shift+k move up
bindsym $mod+Shift+l move right

# Cambiar modo de disposición de la ventana (tiling, floating, etc)
bindsym $mod+f fullscreen toggle
bindsym $mod+Shift+space floating toggle

# Ajustes de layout
bindsym $mod+space focus mode_toggle
bindsym $mod+s split h
bindsym $mod+v split v

# Subir volumen (usa pactl, asegúrate de tener pulseaudio o pipewire)
bindsym XF86AudioRaiseVolume exec --no-startup-id pactl set-sink-volume @DEFAULT_SINK@ +5%

# Bajar volumen
bindsym XF86AudioLowerVolume exec --no-startup-id pactl set-sink-volume @DEFAULT_SINK@ -5%

# Silenciar/activar sonido
bindsym XF86AudioMute exec --no-startup-id pactl set-sink-mute @DEFAULT_SINK@ toggle

# Subir brillo (requiere xbacklight o brillo compatible)
bindsym XF86MonBrightnessUp exec --no-startup-id xbacklight -inc 10

# Bajar brillo
bindsym XF86MonBrightnessDown exec --no-startup-id xbacklight -dec 10

# Opcional: bloquear pantalla con i3lock
bindsym $mod+Shift+x exec i3lock -c 000000

# Status bar (si usas i3status o i3blocks, ajustar según prefieras)
bar {
    status_command i3status
    position top
    font pango:Iosevka Nerd Font 12
}

# Configuración extra, si necesitas:
# Especificar workspace (espacios de trabajo)
set $ws1 "1: "
set $ws2 "2: "
set $ws3 "3: "
set $ws4 "4: "

bindsym $mod+1 workspace $ws1
bindsym $mod+2 workspace $ws2
bindsym $mod+3 workspace $ws3
bindsym $mod+4 workspace $ws4
